}

using System;
using System.Collections.Generic;
using System.Linq;
using Runtime.NovaForge.Economy;
using Runtime.NovaForge.Factions;

namespace Runtime.NovaForge.Missions;

public sealed class MissionRuntimeService : IMissionRuntimeService
{
private readonly Dictionary<string, MissionRuntimeState> _missions = new();
private readonly IEconomyRuntimeService _economy;
private readonly IFactionRuntimeService _factions;

public MissionRuntimeService(IEconomyRuntimeService economy, IFactionRuntimeService factions)
{
    _economy = economy;
    _factions = factions;
}

public MissionRuntimeState OfferMission(string missionId)
{
    var state = EnsureMission(missionId);
    state.Status = MissionStatus.Offered;
    state.UpdatedAtUtc = DateTimeOffset.UtcNow;
    return state;
}

public MissionRuntimeState AcceptMission(string missionId)
{
    var state = EnsureMission(missionId);
    state.Status = MissionStatus.Active;
    state.UpdatedAtUtc = DateTimeOffset.UtcNow;
    return state;
}

public MissionRuntimeState CompleteObjective(string missionId, string objectiveId)
{
    var state = EnsureMission(missionId);
    var objective = state.Objectives.FirstOrDefault(o => o.ObjectiveId == objectiveId);
    if (objective != null) objective.IsComplete = true;
    state.UpdatedAtUtc = DateTimeOffset.UtcNow;
    return state;
}

public MissionRuntimeState CompleteMission(string missionId)
{
    var state = EnsureMission(missionId);
    state.Status = MissionStatus.Completed;
    state.UpdatedAtUtc = DateTimeOffset.UtcNow;

    _economy.ApplySupplyDelta("station_orbit_12_market", "hull_plate_mk1_a", 6);
    _factions.ApplyStandingDelta("industrial_union", 4, "player_completed_union_salvage_contract");
    return state;
}

public MissionRuntimeState FailMission(string missionId, string reason)
{
    var state = EnsureMission(missionId);
    state.Status = MissionStatus.Failed;
    state.UpdatedAtUtc = DateTimeOffset.UtcNow;
    _factions.ApplyStandingDelta("industrial_union", -2, $"mission_failed:{reason}");
    return state;
}

public MissionRuntimeState? GetMission(string missionId) => _missions.TryGetValue(missionId, out var state) ? state : null;

private MissionRuntimeState EnsureMission(string missionId)
{
    if (_missions.TryGetValue(missionId, out var state)) return state;

    state = new MissionRuntimeState
    {
        MissionId = missionId,
        Status = MissionStatus.Generated,
        Objectives = new List<MissionObjectiveState>
        {
            new() { ObjectiveId = "reach_wreck", ObjectiveType = "reach", TargetId = "wreck_lane_a" },
            new() { ObjectiveId = "recover_part", ObjectiveType = "retrieve", TargetId = "hull_plate_mk1_a" },
            new() { ObjectiveId = "return_loot", ObjectiveType = "deliver", TargetId = "dev_storage_alpha" }
        },
        Rewards = new MissionRewardState
        {
            Credits = 1200,
            Items = new List<string> { "hull_plate_mk1_a" },
            Reputation = new Dictionary<string, int> { ["industrial_union"] = 4 }
        }
    };
    _missions[missionId] = state;
    return state;
}
}

using System;
using System.Collections.Generic;

namespace Runtime.NovaForge.Economy;

public sealed class EconomyRuntimeService : IEconomyRuntimeService
{
private readonly Dictionary<string, MarketNodeRuntimeState> _nodes = new();

public MarketNodeRuntimeState GetOrCreateNode(string nodeId)
{
    if (_nodes.TryGetValue(nodeId, out var node)) return node;

    node = new MarketNodeRuntimeState
    {
        NodeId = nodeId,
        NodeType = "trade_hub",
        FactionOwner = "industrial_union",
        Items = new Dictionary<string, ItemMarketRuntimeState>
        {
            ["hull_plate_mk1_a"] = new() { ItemId = "hull_plate_mk1_a", BasePrice = 180, LocalSupply = 40, LocalDemand = 52 },
            ["sealant_cartridge"] = new() { ItemId = "sealant_cartridge", BasePrice = 60, LocalSupply = 18, LocalDemand = 35, PriceMultiplier = 1.15 }
        }
    };
    _nodes[nodeId] = node;
    return node;
}

public void ApplySupplyDelta(string nodeId, string itemId, int delta)
{
    var node = GetOrCreateNode(nodeId);
    if (!node.Items.TryGetValue(itemId, out var item))
    {
        item = new ItemMarketRuntimeState { ItemId = itemId, BasePrice = 100, LocalSupply = 0, LocalDemand = 0 };
        node.Items[itemId] = item;
    }

    item.LocalSupply = Math.Max(0, item.LocalSupply + delta);
    item.PriceMultiplier = RecalculatePrice(nodeId, itemId);
}

public double RecalculatePrice(string nodeId, string itemId)
{
    var node = GetOrCreateNode(nodeId);
    var item = node.Items[itemId];
    var pressure = (item.LocalDemand - item.LocalSupply) / Math.Max(1.0, item.LocalDemand + item.LocalSupply);
    var multiplier = 1.0 + pressure;
    multiplier = Math.Clamp(multiplier, 0.6, 1.8);
    item.PriceMultiplier = Math.Round(multiplier, 2);
    return item.PriceMultiplier;
}
}

using System.Collections.Generic;

namespace Runtime.NovaForge.Factions;

public sealed class FactionRuntimeService : IFactionRuntimeService
{
private readonly Dictionary<string, FactionStandingRuntimeState> _standings = new();

public FactionStandingRuntimeState GetOrCreateStanding(string factionId)
{
    if (_standings.TryGetValue(factionId, out var standing)) return standing;

    standing = new FactionStandingRuntimeState { FactionId = factionId, Standing = 0 };
    _standings[factionId] = standing;
    return standing;
}

public void ApplyStandingDelta(string factionId, int delta, string eventId)
{
    var standing = GetOrCreateStanding(factionId);
    standing.Standing += delta;
    standing.RecentEvents.Add(eventId);
    if (standing.RecentEvents.Count > 12)
    {
        standing.RecentEvents.RemoveAt(0);
    }
}
}
